+def is_palindrome(text: str) -> bool:
+    """
+    Checks if a given text is a palindrome, ignoring case and spaces.
+    """
+    normalized_text = "".join(char.lower() for char in text if char.isalnum())
+    return normalized_text == normalized_text[::-1]
